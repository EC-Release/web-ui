type System ¶ 

type System struct {
    Alloc      float64 //`json:"Alloc"` // bytes allocated and not yet freed
    TotalAlloc float64 //`json:"TotalAlloc"`// bytes allocated (even if freed)
    Sys        float64 //`json:"Sys"`// bytes obtained from system (sum of XxxSys below)
    Lookups    float64 //`json:"Lookups"`// number of pointer lookups
    Mallocs    float64 //`json:"Mallocs"`// number of mallocs
    Frees      float64 //`json:"Frees"`// number of frees

    // Main allocation heap statistics.
    HeapAlloc    float64 //`json:"HeapAlloc"`// bytes allocated and not yet freed (same as Alloc above)
    HeapSys      float64 //`json:"HeapSys"`// bytes obtained from system
    HeapIdle     float64 //`json:"HeapIdle"`// bytes in idle spans
    HeapInuse    float64 //`json:"HeapInuse"`// bytes in non-idle span
    HeapReleased float64 //`json:"HeapReleased"`// bytes released to the OS
    HeapObjects  float64 //`json:"HeapObjects"`// total number of allocated objects
    Report       *Report //`json:"report"`
    Version      string  //`json:"version"`
    IP           string  //`json:"ip"`
    RefID        string
    PID          int    //`json:"pId"`
    AID          string //`json:"AgentId"`
}
