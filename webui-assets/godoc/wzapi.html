type AgentHTTPIntr ¶

Agent HTTP Interface. Each of func below will be invoked during the API mode process. 

type AgentHTTPIntr interface {
    //Launch(optId string)
    //Token(w http.ResponseWriter, r *http.Request, token, optId string)
    InitOps(w http.ResponseWriter, r *http.Request, optId string)
    Status(w http.ResponseWriter, r *http.Request, optId string)
    Hire(w http.ResponseWriter, r *http.Request, optId string)
    Fire(w http.ResponseWriter, r *http.Request, optId string)
    Suspend(w http.ResponseWriter, r *http.Request, optId string)
    Resume(w http.ResponseWriter, r *http.Request, optId string)
    POSTConfig(w http.ResponseWriter, r *http.Request, optId string)
    PUTConfig(w http.ResponseWriter, r *http.Request, optId string)
    GETConfig(w http.ResponseWriter, r *http.Request, optId string)

    //SessionEnd similar to GC invoked by the endpoint /<rev>/<app>/logout for the purpose of managing app resource. The Logout process will stop if the return error is not nil
    SessionEnd(r *http.Request, tknStatus *model.TokenStatus) error

    // Ext extend the Agent API in the following format-
    //<agent_url>/<rev. e,g v1.1|v1.1beta>/<app name from -app flag>/custom/path/name
    //this api extension should assume that the user has been authenticated and promoted by the security chain.
    Ext(w http.ResponseWriter, r *http.Request, tknStatus *model.TokenStatus)
}
