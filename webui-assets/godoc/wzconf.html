type Config ¶

basic agent setting structure. 

type Config struct {
    //in HTTP format<http-schema://host:port>.
    CompressionLevel int    `json:"compressionLevel"`
    FileToUpload     string `json:"-"`
    FileToDownload   string `json:"-"`
    OperationType    string `json:"op"`

    //in TCP format<host:port>
    ResourceHost  string `json:"-"`
    ResourcePorts string `json:"-"`

    //reserved for VLAN-related usage
    ResourceAddress string `json:"resourceAddress"`

    //reserved for VLAN-related usage
    ResourceNetwork string `json:"resourceNetwork"`

    //indicate VLAN. When set in server mode, all traffic from the client will redirect to the designated ip/port based on the VLAN plugin setting. Please refer to VLAN plugin doc.
    VLAN bool `json:vlan`

    //indicate google pprof profiling. when set to true, the subsequence detail will be available in the format <agent-http-endpoint>/911/<default-pprof-URI>. Please refer to google pprof for the possible URIs for the usage.
    //deprecated
    //A911                     bool     `json:"-"`
    Id         string `json:"id"`
    OAuthToken string `json:"oauthToken"`
    BasicToken string `json:"basicToken"`

    //indicate the connectivity mode. Avaialble in server|client|gateway|gw-server|gw-client.
    ClientType string `json:"clientType"`

    //reserved for connectivity session
    BindId string `json:"bindId"`

    TargetServerId string `json:"targetServerId"`
    LocalPort      string `json:"-"`
    FuseModePort   string `json:"-"`
    HealthPort     string `json:"-"`
    HealthAuth     bool   `json:"-"`

    Version string `json:"version"`

    //reserved for OAuth2 provider, and for the health api Basic Auth in this http-header format "Authentication:Basic  base64<ClientId:ClientSecret>". Deprecated in the future cert-based release
    ClientId string `json:"-"`

    //reserved for OAuth2 providers, and for the health api Basic Auth in this http-header format "Authentication:Basic  base64<ClientId:ClientSecret>". Deprecated in the future cert-based release
    ClientSecret string `json:"-"`

    //indicate a http proxy for  connectivity if needed.
    Proxy url.URL `json:"-"`

    GatewayHost url.URL `json:"ghost"`
    HST_NPX     bool    `json:"-"`

    //reserved for OAuth2 providers. Deprecated in the future cert-based release
    OAuthURL url.URL `json:"-"`
    OA2_NPX  bool    `json:"-"`

    //endpoint for reporting purposes. Please refer to the wiki for possible endpoints.
    ServiceURL url.URL `json:"-"`
    SST_NPX    bool    `json:"-"`

    //cloud foundry service id. Deprecated in future cert-based releases.
    Zone      string   `json:"-"`
    GroupID   string   `json:"groupId"`
    Duration  int64    `json:"-"`
    WhiteList []string `json:"-"`
    BlockList []string `json:"-"`

    //cert update
    Cert      *util.Cert `json:"-"`
    TLSKeyDir string     `json:"-"`
    TLSCrtDir string     `json:"-"`

    //the report schema rendered in the healthcheck endpoint.
    Report      *model.Report `json:"report"`
    AdminId     string        `json:"-"`
    AdminSecret string        `json:"-"`
    Debug       bool          `json:"-"`
    Ip          []net.IP      `json:"ip"`

    //reserved for cloud foundry container access
    GatewayID string `json:"-"`

    //reserved for cloud foundry container access
    ReferenceID string `json:"refId"`

    PublicKey string `json:"-"`
    // contains filtered or unexported fields
}
